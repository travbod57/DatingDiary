public string HomeViewItemDescription
{
    get
    {
        if (this.Age != null && this.Town != null)
            return string.Format("Age {0}, {1}", this.Age, this.Town);
        else if (this.Age != null)
            return this.Age.ToString();
        else if (this.Age != null)
            return this.Town;
        else
            return string.Empty;
    }
}
